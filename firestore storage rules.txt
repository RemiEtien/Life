rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {

    // --- Аватары ---
    // Разрешает чтение любому аутентифицированному пользователю.
    // Разрешает запись (создание/обновление/удаление) только владельцу аватара.
    // Дополнительно проверяет, что файл является изображением и его размер менее 5 МБ.
    match /avatars/{userId} {
      allow read: if request.auth != null;
      allow write: if request.auth != null 
                   && request.auth.uid == userId
                   && request.resource.size < 5 * 1024 * 1024
                   && request.resource.contentType.matches('image/.*');
    }

    // --- Файлы воспоминаний ---
    // Разрешает пользователю полный доступ (чтение, запись) только к файлам
    // внутри его собственной директории /users/{userId}/
    match /users/{userId}/{allPaths=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
  }
}
